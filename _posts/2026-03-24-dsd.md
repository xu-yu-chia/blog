module MUX(out,a,b,sel);
    output out;
    input a,b,sel;

    end

endmodule

### 有reset
#### Synchronous reset
![image](https://hackmd.io/_uploads/HkP2yj1i-x.png)
```verilog=
module FF_PN(clk,reset,q,d)
    input reset,clk,d;
    output reg q;
    always @(posedge clk) begin
        if(!reset)
            q <= 0;
        else
            q <= d;
    end
endmodule
```
#### Asynchronous reset
![image](https://hackmd.io/_uploads/HJOqxoks-g.png)

```verilog=
module FF_PN(clk,reset,q,d)
    input reset,clk,d;
    output reg q;
    always @(posedge clk or negedge reset ) begin
        if(!reset)
            q <= 0;
        else
            q <= d;
    end
endmodule
```

## counter
```verilog=
module counter (clk,reset,load,in,count)
    // load 直接從當前值直接到in
    input clk,reset,load;
    output [2:0] in;
    output reg [2:0] count;
    
    always @(posedge clk) begin
        if(reset)
            count <= 3'b000;
        else if (load)
            count <= in;
        else
            count <= count + 1'b1;
    end
endmodule

```

##  frequency divider
用counter來把高頻變成低頻
```verilog=
module frediv(out,clk)
    //輸出頻率 = 輸入頻率 / (2N)
    input clk;
    output out;
    reg [31:0] count=0,N;


    always @(posedge clk) begin
        if (count == N-1) begin
            count <= 0;
            out <= ~out;
        end 
        else begin
            count <= count + 1;
        end
    end
```

## synchronous FIFO
![image](https://hackmd.io/_uploads/H1h12n1i-x.png)

```verilog=
module sync_fifo #(
    parameter DATA_WIDTH = 8,
    parameter DEPTH = 16,
    parameter ADDR_WIDTH = 4   // because 2^4 = 16
)(
    input                       clk,
    input                       reset,
    input                       wen,
    input                       ren,
    input      [DATA_WIDTH-1:0] din,
    output reg [DATA_WIDTH-1:0] dout,
    output                      full,
    output                      empty
);

    // FIFO memory
    reg [DATA_WIDTH-1:0] mem [0:DEPTH-1]; //內部記憶體

    // read / write pointers
    reg [ADDR_WIDTH-1:0] w_ptr;
    reg [ADDR_WIDTH-1:0] r_ptr;

    // count how many data items are currently stored
    reg [ADDR_WIDTH:0] count;

    // status flags
    assign empty = (count == 0);
    assign full  = (count == DEPTH);

    always @(posedge clk or posedge reset) begin
        if (reset) begin
            w_ptr <= 0;
            r_ptr <= 0;
            count <= 0;
            dout  <= 0;
        end 
        else begin
            case ({wen && !full, ren && !empty})

                // 2'b10 : write only
                2'b10: begin
                    mem[w_ptr] <= din; //data in
                    w_ptr <= (w_ptr == DEPTH-1) ? 0 : w_ptr + 1;
                    count <= count + 1;
                end

                // 2'b01 : read only
                2'b01: begin
                    dout <= mem[r_ptr];
                    r_ptr <= (r_ptr == DEPTH-1) ? 0 : r_ptr + 1;
                    count <= count - 1;
                end

                // 2'b11 : write and read at the same time
                2'b11: begin
                    mem[w_ptr] <= din;
                    dout <= mem[r_ptr];

                    w_ptr <= (w_ptr == DEPTH-1) ? 0 : w_ptr + 1;
                    r_ptr <= (r_ptr == DEPTH-1) ? 0 : r_ptr + 1;

                    // count does not change
                    count <= count;
                end

                // 2'b00 : no operation
                default: begin
                    w_ptr <= w_ptr;
                    r_ptr <= r_ptr;
                    count <= count;
                    dout  <= dout;
                end
            endcase
        end
    end

endmodule
```
